select seq as prime
from sequence as devidend
where seq > 1
and not exists
	(select *
	from sequence as devisor
	where devisor.seq <= devidend.seq / 2
	and devisor.seq <> 1
	and mod(devidend.seq ,devisor.seq) = 0
	)
order by prime
